拆一個C宣告

int *(*(*(**x[])(char*, int*(*)(char*)))[])(char**, char*(*)());

從meme拆出來的，那麼稍微的讀一下好了

x 是一個陣列，存了指向函式指標的指標

這個函式指標接受一個char指標，和另外一個接受一個char指標回傳一個int指標的函式指標

其回傳一個指向陣列的指標，陣列裏面存的還是一個函式指標

這個函式指標接受一個char雙重指標，和一個無參並回傳一個char指標的函式指標

其回傳一個int指標

用法大概是送一個字串和一個轉換函式，拿到一組行爲，再挑一個送資料和輔助函式，拿到int*

其集合了工廠，策略表和結果三位一體

非常不錯，愛來自BiFunction<String, Function<Function<Integer, Runnable[]>[]>, Function<String[], Function<String, Integer>[]>>[][] x = null;

但唯一的問題就是寫這個東西的人最好只是寫着玩，或者真的需要兩段式決策帶來的動態轉圜空間

否則就有一點點的不是人類

另外如果例如存在複數個工廠導引表（在某種複雜的外掛系統）

其實可以改進成

    int*(*(*(**x[])(char*, const int* (*const (*const volatile (*)(int))[])(const char*)))[])(char**, char*(*)());

加了一個函數陣列，每個函數接受一個 唯讀char*，返回 唯讀int*，而這個函數陣列本身是另一個函數的唯讀返回值

這個外層函數指標易變動，強制每次從記憶體讀取並唯讀

其接受一個 int 作參數，實際上是開始用數字選導引表本身

變成三段式動態決策，且一段決策在高危環境

當然導引表，策略表這兩個是現說的，有相近的概念自動對應即可

雖然這種程度的大解耦是否值得有待商榷，但可以肯定的是絕對不可能出現在協作環境

實際測試下來確實在功能上非常完美，只要不去看那一串東西，該做的都做了:

#include <stdio.h>
#include <stdlib.h>

int global_val = 42;
int* final_target(char** a, char*(*b)()) {
    printf("%s Says: %s\n",*a,b());
    return &global_val;
}

int* (*level2_array[2])(char**, char*(*)()) = { final_target, NULL };

/*
 * 這一個系列都沒被invoke，佔位用,雖然也可以invoke
*/
const int* inner_func(const char* s) { return NULL; }
const int* (*inner_ptr_array[1])(const char*) = { inner_func };
const int* (*const (*const volatile complex_arg_func(int n))[])(const char*) {
    return &inner_ptr_array;
}

int* (*(*target_func(char* s, const int* (*const (*const volatile (p)(int))[])(const char*)))[])(char**, char*(*)()) {
    /*　這裏就可以動 msg s 和那個 complex_arg_func(int) 
     *  比如什麼 const int *const result = (*(*complex_arg_func(1))[0])("何意味");
     * */
    return &level2_array;
}

int* (*(*(*ptr_to_target_func)(char*, const int* (*const (*const volatile (*)(int))[])(const char*)))[])(char**, char*(*)()) = target_func;
int* (*(*(*(*ptr_to_ptr_to_target_func))(char*, const int* (*const (*const volatile (*)(int))[])(const char*)))[])(char**, char*(*)()) = &ptr_to_target_func;

char* callback_impl() { return "桜小路ルナ様"; }

int main() {
    /* 萬物起源這一塊 */
    int* (*(*(**x[1])(char*, const int* (*const (*const volatile (*)(int))[])(const char*)))[])(char**, char*(*)());

    x[0] = ptr_to_ptr_to_target_func;

    char* msg = "Texas0295";
    char** msg_ptr = &msg;

    /*
     * x[0]解一次，*x[0]解一次拿到target_func ->　一次週轉拿工廠導引表
     * 現在執行那個選定的工廠target_func，傳一個msg和comple_arg_func。最後拿到一個level2_array ->　二次週轉拿策略表
     * 寫0拿函式，傳msg_ptr,和callback_impl給那個函式 -> 三次週轉執行選定策略
     * 最後拿到一個42
    */
    int* result = (*(*(*x[0])(msg, complex_arg_func))[0])(msg_ptr, callback_impl);

    if (result) {
        printf("Result: %d\n", *result);
    }

    return 0;
}
